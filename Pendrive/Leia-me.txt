#Mudar senha de Windows
cd /mnt/windows/Windows/System32/config
- Listar os usuários: sudo chntpw -l SAM
- Exemplo de mudança de usuário: sudo chntpw -u jose SAM

#Exemplo de teste em SSD NVME
sudo smartctl -H -i -c -A -l error -l selftest /dev/nvme0n1

#Exemplo de teste rápido em discos ATA
sudo smartctl -H -i -c -A -l error -l selftest -l selective /dev/sda

#Mostrar status dos seus dispositivos de armazenamento
sudo pkexec gsmartcontrol

#Exemplo de Scan de badblocks sem destrução de dados
sudo badblocks -nsv -c 10240 /dev/sda1

#Verificação rápida de discos
sudo QDiskInfo

#Clonagem e backup de discos
sudo clonezilla
sudo rescuezilla

#Editar partições
sudo gparted

#TESTES DA PLACA DE VÍDEO

#Modo padrão
glmark2

#Se quiser rodar o teste em _fullscreen:
glmark2 --fullscreen

#Para rodar somente um checklist rápido de vídeo:
glmark2 --validate

#E se quiser todos os detalhes do teste, rode no modo _debug:
glmark2 -d

#Recuperando partições deletadas
sudo testdisk

#Recuperando arquivos deletados
sudo photorec
